    .replace(/```[\s\S]*?```/g, m => '<code class="cmd">' + m.replace(/```/g,'').trim() + '</code>')

async function sendMessage() {
  if (isWaiting) return;
  const input = document.getElementById('msg-input');
  const text = input.value.trim();
  if (!text) return;
  input.value = ''; input.style.height = '';
  addMsg('user', text);
  isWaiting = true; setStatus(true); showBanner(false);

  const msgs = document.getElementById('messages');
  const row = document.createElement('div');
  row.className = 'msg-row aura'; row.id = 'stream-row';
  row.innerHTML =
    '<div class="msg-avatar">' + CROWN_Q + '</div>' +
    '<div class="msg-wrap">' +
      '<div class="bubble" id="stream-bubble"><span class="typing-cursor"></span></div>' +
      '<div class="msg-time">' + now() + '</div>' +
    '</div>';
  msgs.appendChild(row);
  msgs.scrollTop = msgs.scrollHeight;

  let tamCevap = '';
  try {
    const res = await fetch(API_URL + '/api/stream', {
      method:'POST', headers:{'Content-Type':'application/json'},
      body:JSON.stringify({ mesaj: text })
    });
    if (!res.ok) throw new Error('HTTP ' + res.status);
    const reader = res.body.getReader();
    const decoder = new TextDecoder();
    while (true) {
      const { done, value } = await reader.read();
      if (done) break;
      const chunk = decoder.decode(value);
      for (const line of chunk.split('\n')) {
        if (!line.startsWith('data: ')) continue;
        try {
          const data = JSON.parse(line.slice(6));
          const bubble = document.getElementById('stream-bubble');
          if (!bubble) continue;
          if (data.token) {
            tamCevap += data.token;
            let g = tamCevap.replace(/\n/g,'<br>').replace(/`([^`]+)`/g,'<code class="cmd">$1</code>');
            bubble.innerHTML = g + '<span class="typing-cursor"></span>';
            msgs.scrollTop = msgs.scrollHeight;
          }
          if (data.done) {
            let g = tamCevap.replace(/\n/g,'<br>').replace(/`([^`]+)`/g,'<code class="cmd">$1</code>');
            bubble.innerHTML = g;
          }
        } catch(e) {}
      }
    }
    showBanner(false);
  } catch (err) {
    const bubble = document.getElementById('stream-bubble');
    if (bubble) bubble.innerHTML = '⚠ Bağlantı hatası<br><small>' + err.message + '</small>';
    showBanner(true);
  } finally {
    const r = document.getElementById('stream-row'); if (r) r.removeAttribute('id');
    const b = document.getElementById('stream-bubble'); if (b) b.removeAttribute('id');
    isWaiting = false; setStatus(false);
  }
}

(function() {
  var el = document.getElementById('quick');
  if (!el) return;
  var isDown = false, startX, scrollLeft;
  el.addEventListener('mousedown', function(e) { isDown=true; el.classList.add('grabbing'); startX=e.pageX-el.offsetLeft; scrollLeft=el.scrollLeft; });
  el.addEventListener('mouseleave', function() { isDown=false; el.classList.remove('grabbing'); });
  el.addEventListener('mouseup', function() { isDown=false; el.classList.remove('grabbing'); });
  el.addEventListener('mousemove', function(e) { if (!isDown) return; e.preventDefault(); el.scrollLeft=scrollLeft-(e.pageX-el.offsetLeft-startX)*1.5; });
  el.addEventListener('wheel', function(e) { e.preventDefault(); el.scrollLeft+=e.deltaY*0.8; }, { passive:false });
})();

function konumGonder() {
  if (!navigator.geolocation) { addMsg('aura', 'Tarayıcı konum desteklemiyor.'); return; }
  navigator.geolocation.getCurrentPosition(
    function(pos) {
      var lat = pos.coords.latitude.toFixed(6);
      var lon = pos.coords.longitude.toFixed(6);
      var acc = Math.round(pos.coords.accuracy);
      var url = 'https://maps.google.com/?q=' + lat + ',' + lon;
      addMsg('user', 'Konumumu paylaşıyorum');
      addMsg('aura', '📍 Konum alındı!<br>🗺️ <a href="' + url + '" target="_blank" style="color:var(--accent3)">Google Maps</a><br>📡 Doğruluk: ' + acc + 'm<br>📌 ' + lat + ', ' + lon);
    },
    function(err) { addMsg('aura', 'Konum alınamadı: ' + err.message); },
    { enableHighAccuracy:true, timeout:10000 }
  );
}
