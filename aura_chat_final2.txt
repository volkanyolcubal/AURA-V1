// ── AYARLAR ─────────────────────────────────────────────────────
localStorage.removeItem('aura_url');
let API_URL = window.location.origin;
let isWaiting = false;

// Sayfa yüklenince sıfırla
window.addEventListener('focus', function() { isWaiting = false; setStatus(false); });

function saveSettings() {
  const val = document.getElementById('api-url-input').value.trim();
  if (val) {
    API_URL = val.replace(/\/$/, '');
    localStorage.setItem('aura_url', API_URL);
  }
  closeSettings();
  addMsg('aura', `✅ Adres güncellendi: ${API_URL}`);
}
function openSettings() {
  document.getElementById('api-url-input').value = API_URL;
  document.getElementById('settings-modal').classList.add('open');
}
function closeSettings() {
  document.getElementById('settings-modal').classList.remove('open');
}
function closeSettingsOutside(e) {
  if (e.target === document.getElementById('settings-modal')) closeSettings();
}

// ── ZAMAN ────────────────────────────────────────────────────────
function now() {
  return new Date().toLocaleTimeString('tr-TR', { hour: '2-digit', minute: '2-digit' });
}

// ── MESAJ EKLE ───────────────────────────────────────────────────
function addMsg(type, text) {
  const msgs = document.getElementById('messages');

  // kod bloklarını <code> yap
  text = text
    .replace(/```[\s\S]*?```/g, m => `<code class="cmd">${m.replace(/```/g,'').trim()}</code>`)
    .replace(/`([^`]+)`/g, '<code class="cmd">$1</code>')
    .replace(/\n/g, '<br>');

  const row = document.createElement('div');
  row.className = `msg-row ${type}`;

  const avatarTxt = type === 'aura' ? 'AV' : 'VK';
  row.innerHTML = `
    <div class="msg-avatar">${avatarTxt}</div>
    <div>
      <div class="bubble">${text}</div>
      <div class="msg-time">${now()}</div>
    </div>
  `;
  msgs.appendChild(row);
  msgs.scrollTop = msgs.scrollHeight;
}

// ── TİPİNG GÖSTERGESİ ────────────────────────────────────────────

// ── GÖNDER ───────────────────────────────────────────────────────
async function sendMessage() {
  if (isWaiting) return;
  const input = document.getElementById('msg-input');
  const text = input.value.trim();
  if (!text) return;

  input.value = '';
  input.style.height = '';
  addMsg('user', text);

  isWaiting = true;
  setStatus(true);
  showBanner(false);

  // Streaming bubble
  const msgs = document.getElementById('messages');
  const row = document.createElement('div');
  row.className = 'msg-row aura';
  row.id = 'stream-row';
  row.innerHTML = `
    <div class="msg-avatar">AV</div>
    <div>
      <div class="bubble" id="stream-bubble"><span class="typing-cursor">&#9611;</span></div>
      <div class="msg-time">${now()}</div>
    </div>
  `;
  msgs.appendChild(row);
  msgs.scrollTop = msgs.scrollHeight;

  let tamCevap = '';

  try {
    const res = await fetch(`${API_URL}/api/stream`, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ mesaj: text }),
    });

    if (!res.ok) throw new Error(`HTTP ${res.status}`);

    const reader = res.body.getReader();
    const decoder = new TextDecoder();

    while (true) {
      const { done, value } = await reader.read();
      if (done) break;

      const chunk = decoder.decode(value);
      const lines = chunk.split('\n');

      for (const line of lines) {
        if (!line.startsWith('data: ')) continue;
        try {
          const data = JSON.parse(line.slice(6));
          const bubble = document.getElementById('stream-bubble');
          if (!bubble) continue;
          if (data.token) {
            tamCevap += data.token;
            let g = tamCevap.replace(/\n/g, '<br>').replace(/`([^`]+)`/g, '<code class="cmd">$1</code>');
            bubble.innerHTML = g + '<span class="typing-cursor">&#9611;</span>';
            msgs.scrollTop = msgs.scrollHeight;
          }
          if (data.done) {
            let g = tamCevap.replace(/\n/g, '<br>').replace(/`([^`]+)`/g, '<code class="cmd">$1</code>');
            bubble.innerHTML = g;
          }
        } catch(e) {}
      }
    }
    showBanner(false);

  } catch (err) {
    const bubble = document.getElementById('stream-bubble');
    if (bubble) bubble.innerHTML = `⚠️ Bağlantı hatası<br><small>${err.message}</small>`;
    showBanner(true);
  } finally {
    const r = document.getElementById('stream-row');
    if (r) r.removeAttribute('id');
    const b = document.getElementById('stream-bubble');
    if (b) b.removeAttribute('id');
    isWaiting = false;
    setStatus(false);
  }
}

function quickSend(cmd) {
  document.getElementById('msg-input').value = cmd;
  sendMessage();
}

// ── INPUT ────────────────────────────────────────────────────────
function handleKey(e) {
  if (e.key === 'Enter' && !e.shiftKey) {
    e.preventDefault();
    sendMessage();
  }
}
function autoResize(el) {
  el.style.height = 'auto';
  el.style.height = Math.min(el.scrollHeight, 120) + 'px';
}



// ── YATAY SCROLL (FARE) ──────────────────────────────────────
(function() {
  var el = document.getElementById('quick');
  if (!el) return;
  var isDown = false, startX, scrollLeft;
  el.addEventListener('mousedown', function(e) {
    isDown = true; el.classList.add('grabbing');
    startX = e.pageX - el.offsetLeft;
    scrollLeft = el.scrollLeft;
  });
  el.addEventListener('mouseleave', function() { isDown = false; el.classList.remove('grabbing'); });
  el.addEventListener('mouseup', function() { isDown = false; el.classList.remove('grabbing'); });
  el.addEventListener('mousemove', function(e) {
    if (!isDown) return;
    e.preventDefault();
    var x = e.pageX - el.offsetLeft;
    el.scrollLeft = scrollLeft - (x - startX) * 1.5;
  });
  // Fare tekerleği ile de kaydır
  el.addEventListener('wheel', function(e) {
    e.preventDefault();
    el.scrollLeft += e.deltaY * 0.8;
  }, { passive: false });
})();

function konumGonder() {
  if (!navigator.geolocation) {
    addMsg('aura', 'Tarayici konum desteklemiyor.');
    return;
  }
  navigator.geolocation.getCurrentPosition(
    function(pos) {
      var lat = pos.coords.latitude.toFixed(6);
      var lon = pos.coords.longitude.toFixed(6);
      var acc = Math.round(pos.coords.accuracy);
      var url = 'https://maps.google.com/?q=' + lat + ',' + lon;
      addMsg('user', 'Konumumu paylaşıyorum');
      addMsg('aura', '📍 Konum alındı!<br>🗺️ <a href="' + url + '" target="_blank" style="color:var(--accent)">Google Maps</a><br>📡 Doğruluk: ' + acc + 'm<br>📌 ' + lat + ', ' + lon);
    },
    function(err) {
      addMsg('aura', 'Konum alinamadi: ' + err.message);
    },
    { enableHighAccuracy: true, timeout: 10000 }
  );
}

function clearChat() {

    document.getElementById('messages').innerHTML = '';
    addMsg('aura', '🗑️ Ekran temizlendi. AURA-V hafızası korunuyor.');
  }
}
